// -*- C -*-
uint32_t u, i, t;
float x, y, U;

while (true) {
  u = draw_uint32(rng);
  i = u >> 24;
  t = u & 0x00FFFFFFu;                 // bits 23..0
  x = t*wehf[i];
  if (t < ke2f[i]) return x;
  
  if (i == 0) { // bottom strip
    U = draw_u01f(rng);
    return r - log1pf(-U);
  }
  // slow path (i > 0):
  y = fef[i] + draw_u01f(rng)*(fef[i-1] - fef[i]);
  if (y < expf(-x)) return x;
}
